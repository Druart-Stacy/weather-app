function fetchWeather(city) {
    document.addEventListener('DOMContentLoaded', function() {
        const apiKey = 'YOUR_API_KEY';
        const city = 'London';
    
        const apiURL = `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${apiKey}&units=metric`;
    
        function getWeather() {
            fetch(apiURL)
                .then(response => response.json())
                .then(data => {
                    // Obtenir les informations météo nécessaires depuis les données reçues
                    const weatherDescription = data.weather[0].description;
                    const iconCode = data.weather[0].icon;
    
                    // Afficher la description météo
                    const weatherDiv = document.getElementById('weather');
                    weatherDiv.innerHTML = `<p>Weather: ${weatherDescription}</p>`;
    
                    // Afficher l'icône correspondant à la météo
                    const iconURL = `http://openweathermap.org/img/wn/${iconCode}.png`;
                    const iconImg = document.createElement('img');
                    iconImg.src = iconURL;
                    weatherDiv.appendChild(iconImg);
                })
                .catch(error => {
                    console.error('Error fetching weather data:', error);
                });
        }
    
        getWeather();
    });
